SET FOREIGN_KEY_CHECKS = 0;

-- ==========================================
-- 1. CREACIÓN DE BASE DE DATOS
-- ==========================================
DROP DATABASE IF EXISTS insuban_db;
CREATE DATABASE insuban_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE insuban_db;

-- ==========================================
-- 2. ESTRUCTURA DE TABLAS
-- ==========================================

-- Tablas Geográficas y Auxiliares
CREATE TABLE paises (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL
);

CREATE TABLE regiones (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL,
    pais_id INT NOT NULL,
    FOREIGN KEY (pais_id) REFERENCES paises(id)
);

CREATE TABLE comunas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL,
    region_id INT NOT NULL,
    FOREIGN KEY (region_id) REFERENCES regiones(id)
);

CREATE TABLE tipos_venta (
    id INT AUTO_INCREMENT PRIMARY KEY,
    descripcion VARCHAR(50) NOT NULL
);

-- Tablas de Sistema y Usuarios
CREATE TABLE roles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE,
    descripcion VARCHAR(100)
);

CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL,
    apellido VARCHAR(50) NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    telefono VARCHAR(20),
    rol_id INT NOT NULL,
    activo BOOLEAN DEFAULT TRUE,
    FOREIGN KEY (rol_id) REFERENCES roles(id)
);

-- Inventario y Activos
CREATE TABLE activos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    codigo_interno VARCHAR(20) UNIQUE,
    nombre VARCHAR(100) NOT NULL,
    descripcion TEXT,
    ubicacion VARCHAR(100)
);

CREATE TABLE categorias_insumo (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL
);

CREATE TABLE insumos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    codigo_sku VARCHAR(50) UNIQUE,
    nombre VARCHAR(150) NOT NULL,
    descripcion TEXT,
    categoria_id INT,
    stock_actual DECIMAL(10,2) DEFAULT 0,
    stock_minimo DECIMAL(10,2) DEFAULT 5,
    unidad_medida VARCHAR(20) DEFAULT 'UN',
    ubicacion_bodega VARCHAR(50),
    FOREIGN KEY (categoria_id) REFERENCES categorias_insumo(id)
);

CREATE TABLE tipos_movimiento (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL,
    factor INT NOT NULL
);

CREATE TABLE movimientos_inventario (
    id INT AUTO_INCREMENT PRIMARY KEY,
    insumo_id INT NOT NULL,
    tipo_movimiento_id INT NOT NULL,
    cantidad DECIMAL(10,2) NOT NULL,
    fecha DATETIME DEFAULT CURRENT_TIMESTAMP,
    usuario_id INT NOT NULL,
    referencia_id INT,
    observacion TEXT,
    FOREIGN KEY (insumo_id) REFERENCES insumos(id),
    FOREIGN KEY (tipo_movimiento_id) REFERENCES tipos_movimiento(id),
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id)
);

-- Compras y Proveedores
CREATE TABLE proveedores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    rut VARCHAR(12) UNIQUE NOT NULL,
    nombre VARCHAR(100) NOT NULL,
    direccion VARCHAR(150),
    email VARCHAR(100),
    telefono VARCHAR(50),
    contacto_vendedor VARCHAR(100),
    tipo_venta_id INT,
    comuna_id INT,
    FOREIGN KEY (tipo_venta_id) REFERENCES tipos_venta(id),
    FOREIGN KEY (comuna_id) REFERENCES comunas(id)
);

CREATE TABLE estados_orden_compra (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL
);

CREATE TABLE ordenes_compra (
    id INT AUTO_INCREMENT PRIMARY KEY,
    proveedor_id INT NOT NULL,
    usuario_creador_id INT NOT NULL,
    fecha_creacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    estado_id INT NOT NULL,
    monto_neto DECIMAL(15,2) DEFAULT 0,
    impuesto DECIMAL(15,2) DEFAULT 0,
    monto_total DECIMAL(15,2) DEFAULT 0,
    FOREIGN KEY (proveedor_id) REFERENCES proveedores(id),
    FOREIGN KEY (usuario_creador_id) REFERENCES usuarios(id),
    FOREIGN KEY (estado_id) REFERENCES estados_orden_compra(id)
);

CREATE TABLE detalle_orden_compra (
    id INT AUTO_INCREMENT PRIMARY KEY,
    orden_compra_id INT NOT NULL,
    insumo_id INT NOT NULL,
    cantidad_solicitada DECIMAL(10,2) NOT NULL,
    cantidad_recibida DECIMAL(10,2) DEFAULT 0,
    precio_unitario DECIMAL(15,2),
    total_linea DECIMAL(15,2),
    FOREIGN KEY (orden_compra_id) REFERENCES ordenes_compra(id),
    FOREIGN KEY (insumo_id) REFERENCES insumos(id)
);

-- Mantenimiento
CREATE TABLE estados_solicitud (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL
);

CREATE TABLE solicitudes_ot (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_solicitante_id INT NOT NULL,
    activo_id INT,
    fecha_solicitud DATETIME DEFAULT CURRENT_TIMESTAMP,
    estado_id INT NOT NULL,
    descripcion_trabajo TEXT,
    FOREIGN KEY (usuario_solicitante_id) REFERENCES usuarios(id),
    FOREIGN KEY (activo_id) REFERENCES activos(id),
    FOREIGN KEY (estado_id) REFERENCES estados_solicitud(id)
);

CREATE TABLE detalle_solicitud (
    id INT AUTO_INCREMENT PRIMARY KEY,
    solicitud_id INT NOT NULL,
    insumo_id INT NOT NULL,
    cantidad DECIMAL(10,2) NOT NULL,
    estado_linea VARCHAR(20) DEFAULT 'PENDIENTE',
    FOREIGN KEY (solicitud_id) REFERENCES solicitudes_ot(id),
    FOREIGN KEY (insumo_id) REFERENCES insumos(id)
);
